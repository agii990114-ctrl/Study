## Javascript 화면제어
```
function load(){
    let html='';

    for(i=0; i<25; i++){
       html += `<div class="bg${(i===6)? 3 : (i %2 === 0)? 2 : 1}"></div>`
    }
    document.getElementsByTagName(`section`)[0].innerHTML = html;
    scValue = true;
}

function clean(){
    document.getElementsByTagName(`section`)[0].innerHTML = "";
    scValue = false;
}

let scValue = false;

function scToggle(){
    (!scValue)? load(): clean();
}
```

1. document : 브라우저 화면상의 정보에 접근
2. getElementsByTagName(`section`)[0] : 브라우저 화면 안에 있는 `section`이라는 태그에 접근; 화면상에 같은 태그가 여러개 있을 가능성이 있으므로 배열로 표시;
3. innerHTML = : 접속한 태그 안에 다음의 내용을 삽입함


## J Query
```
$(() => {
    function view(){
        let html = '';

        for(i=0; i < 25; i++){
            let css = 1;
            if(i%2===0) css =2;
            html += `<div class="bg${css}"></div>`
        };
        $("section").html(html);
        state = true;
    };

    let state = false;

    $("button").off().on("click",(e)=>{
        let index = $("button").index(e.target);
        if(index === 0) {
            view()
        }else if (index === 1){
            $("section").empty();
            state = false
        }else if (index === 2){
            if(!state){
                view()
            }else {
              $("section").empty();  
              state = false
            }
        };
    });
})
```

1. `<script>` src에 제이쿼리 주소 삽입
2. js파일에서 $function(){}으로 제이쿼리 실행 가능 // callback 함수를 사용하여 특정 함수를 제이쿼리 영역으로 사용 가능

```
document.addEventListener('DOMContentLoaded', () => {
    const section = document.querySelector("section");
    const buttons = document.querySelectorAll("button");
    let state = false;

    // 1. 뷰 생성 함수
    const view1 = () => {
        let html = '';
        for (let i = 0; i < 25; i++) {
            // i는 let으로 선언하여 전역 오염 방지
            const bgNum = (i === 6) ? 3 : (i % 2 === 0 ? 2 : 1);
            html += `<div class="bg${bgNum}"></div>`;
        }
        section.innerHTML = html;
        state = true;
    };

    // 2. 이벤트 리스너 설정
    buttons.forEach((button) => {
        button.addEventListener("click", (e) => {
            // e.target이 버튼 내부 요소(span 등)일 수 있으므로 
            // 가장 가까운 button 요소를 찾아 인덱스를 구합니다.
            console.log(e.target);
            const targetButton = e.target.closest("button");
            const index = Array.from(buttons).indexOf(targetButton);

            console.log("클릭된 요소:", e.target);
            console.log("버튼 인덱스:", index);

            if (index === 0) {
                view1();
            } else if (index === 1) {
                section.innerHTML = "";
                state = false;
            } else if (index === 2) {
                if (!state) {
                    view1();
                } else {
                    section.innerHTML = "";
                    state = false;
                }
            }
        });
    });
});
```
